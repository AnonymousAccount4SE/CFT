

P(1,File("Stats2.txt")) => f
Lib.Files.LineReader(f).start => reader
dict=Dict
loop
	line=reader.read
	break(line==null)

	dict.set(line, dict.get(line,0)+1)
|

sum=0
dict.keys->key
	count=dict.get(key)
	sum = sum + count
	out(Int(count, key))
| _.sort->x
	key=x.data
	report(key,dict.get(key),""+(100*dict.get(key)/sum).i+"%")
/run

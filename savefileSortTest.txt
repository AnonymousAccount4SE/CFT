
# Sort files by length (in bytes)
# --
	Dir.files->f
		# Produce list of Int() objects, each wrapping
		# a file, and presenting itself as an int
		# with the value we want to sort on
		out(Int(f.length,f))
	| => list

	# Now we got a list of Int objects, sorting them
	list.sort.reverse => sortedList

	# extract the file objects from the sorted list
	sortedList->x
		out(x.data)
/Sort
